    centos连接wifi无线网 # 

    # list all network devices
$ nmcli dev status

# check if wifi is enabled or not.
$ nmcli radio wifi

# if wifi is disabled, then enable it.
$ nmcli radio wifi on

# 列出附近wifi
$ nmcli dev wifi list

# 连接wifi
nmcli dev wifi connect <network-ssid> [password <password>]
