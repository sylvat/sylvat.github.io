    设置以太网静态IP # 

    # 修改网卡配置
$ vi /etc/sysconfig/network-scripts/ifcfg-enp89s0
# 修改如下：
BOOTPROTO=static
ONBOOT=yes
IPADDR=192.168.0.18
NETMASK=255.255.255.0
GATEWAY=192.168.0.1

# 重启网卡
$ systemctl restart network

    设置wifi连接静态IP # 

    $ vi /etc/sysconfig/network-scripts/ifcfg-TP-LINK_XXXX

# 修改BOOTPROTO为static
BOOTPROTO=static
#开机启用
ONBOOT=yes
# 添加IP
IPADDR=192.168.0.12
NETMASK=255.255.255.0
GATEWAY=192.168.0.1

# 重启
$ reboot

# 查看IP
$ ip addr
